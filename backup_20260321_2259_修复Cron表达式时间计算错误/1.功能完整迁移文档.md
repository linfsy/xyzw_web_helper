// 安全的 localStorage 操作
const safeLocalStorage = {
  getItem: (key) => {
    try {
      return localStorage.getItem(key);
    } catch (error) {
      console.warn('localStorage getItem error:', error);
      return null;
    }
  },
  setItem: (key, value) => {
    try {
      localStorage.setItem(key, value);
      return true;
    } catch (error) {
      console.warn('localStorage setItem error:', error);
      return false;
    }
  },
  removeItem: (key) => {
    try {
      localStorage.removeItem(key);
      return true;
    } catch (error) {
      console.warn('localStorage removeItem error:', error);
      return false;
    }
  },
  clear: () => {
    try {
      localStorage.clear();
      return true;
    } catch (error) {
      console.warn('localStorage clear error:', error);
      return false;
    }
  }
};

// 定期清理任务标记
const cleanupLocalStorageTasks = () => {
  try {
    const keys = Object.keys(localStorage);
    const now = Date.now();
    
    keys.forEach(key => {
      if (key.startsWith('task_executing_')) {
        const value = safeLocalStorage.getItem(key);
        if (value) {
          try {
            const taskData = JSON.parse(value);
            if (taskData.timestamp && (now - taskData.timestamp > 3600000)) {
              safeLocalStorage.removeItem(key);
              console.log('清理过期任务标记:', key);
            }
          } catch (e) {
            safeLocalStorage.removeItem(key);
          }
        }
      }
    });
  } catch (error) {
    console.warn('清理任务标记失败:', error);
  }
};

// 增强的日志记录

      ...log,
      timestamp: new Date().toISOString()
    });
    
    // 只保留最近1000条日志
    if (logs.length > 1000) {
      logs.splice(0, logs.length - 1000);
    }
    
    safeLocalStorage.setItem('task_logs', JSON.stringify(logs));

// 批量执行函数
const executeInBatches = async (taskFunction, taskNames) => {
  if (selectedTokens.value.length === 0) return;
  
  batchTaskStore.startTask();
  shouldStop.value = false;
  
  const totalBatches = Math.ceil(selectedTokens.value.length / batchSettings.batchSize);
  let currentBatch = 1;
  
  for (let i = 0; i < selectedTokens.value.length; i += batchSettings.batchSize) {
    if (shouldStop.value) break;
    
    const batchTokens = selectedTokens.value.slice(i, i + batchSettings.batchSize);
    selectedTokens.value = batchTokens;
    
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 开始执行批次 ${currentBatch}/${totalBatches} (${batchTokens.length}个账号) ===`,
      type: "info",
    });
    
    // 重置 shouldStop，确保任务可以正常执行
    batchTaskStore.resetShouldStop();
    
    // 重要：保存当前的 isRunning 状态
    const wasRunningBeforeTask = batchTaskStore.isRunning;
    
    try {
      await taskFunction();
    } catch (error) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `执行批次时出错: ${error.message}`,
        type: "error",
      });
      console.error('taskFunction error:', error);
    } finally {
      // 重要：恢复 isRunning 状态
      if (wasRunningBeforeTask && !batchTaskStore.isRunning) {
        batchTaskStore.startTask();
      }
    }
    
    // 重要：taskFunction 执行后可能会调用 stopTask() 将 shouldStop 设为 true
    // 必须立即重置 shouldStop，否则下一批会跳过
    batchTaskStore.resetShouldStop();
    
    currentBatch++;
    
    if (i + batchSettings.batchSize < selectedTokens.value.length) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== 批次完成，等待 ${batchSettings.batchInterval} 秒后执行下一批 ===`,
        type: "info",
      });
      await new Promise(resolve => setTimeout(resolve, batchSettings.batchInterval * 1000));
    }
  }
  
  batchTaskStore.stopTask();
};

// 定时任务执行
const executeScheduledTask = async (task) => {
  if (!task) return;
  
  const taskId = task.id;
  const taskExecutingKey = `task_executing_${taskId}`;
  
  // 检查任务是否正在执行
  const isExecuting = safeLocalStorage.getItem(taskExecutingKey);
  if (isExecuting) {
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 任务 ${task.name} 正在执行中，跳过 ===`,
      type: "info",
    });
    return;
  }
  
  // 标记任务开始执行
  safeLocalStorage.setItem(taskExecutingKey, JSON.stringify({
    timestamp: Date.now(),
    taskId: taskId
  }));
  
  try {
    // 检查是否有其他任务在执行
    if (batchTaskStore.isRunning) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== 检测到冲突，任务 ${task.name} 加入积攒列表等待 ===`,
        type: "info",
      });
      
      // 添加到积攒队列
      taskQueue.value.push({
        task: task,
        timestamp: Date.now()
      });
      
      // 保存队列到 localStorage
      safeLocalStorage.setItem('task_queue', JSON.stringify(taskQueue.value));
      
      // 清除任务执行标记
      safeLocalStorage.removeItem(taskExecutingKey);
      return;
    }
    
    // 检查是否在暂停时间
    if (isPauseTime.value.paused) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== 暂停时间，任务 ${task.name} 加入积攒列表等待 ===`,
        type: "info",
      });
      
      // 添加到积攒队列
      taskQueue.value.push({
        task: task,
        timestamp: Date.now()
      });
      
      // 保存队列到 localStorage
      safeLocalStorage.setItem('task_queue', JSON.stringify(taskQueue.value));
      
      // 清除任务执行标记
      safeLocalStorage.removeItem(taskExecutingKey);
      return;
    }
    
    // 执行任务
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 开始执行定时任务: ${task.name} ===`,
      type: "info",
    });

          time: new Date().toLocaleTimeString(),

          type: "success",

        totalSuccess++;
        break;
      } catch (error) {

        let errorMsg = error.message || "未知错误";
        let errorType = "error";

        if (errorMsg.includes("timeout")) {
          errorMsg = "请求超时";
          errorType = "warning";
        } else if (errorMsg.includes("网络")) {
          errorMsg = "网络错误";
          errorType = "warning";
        }

        if (consecutiveErrors <= maxRetries && !shouldStop.value) {
          addLog({
            time: new Date().toLocaleTimeString(),
            message: `=== ${token.name} 领取功法残卷失败: ${errorMsg}，将在3秒后重试 ===`,
