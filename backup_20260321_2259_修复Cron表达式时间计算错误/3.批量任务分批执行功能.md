# 批量任务分批执行功能

## 功能说明

这个功能允许用户在执行批量任务时，将账号分成多个批次执行，每批执行完成后等待自定义时间再执行下一批。

### 主要特性
- ✅ 支持自定义每批账号数量（1-50个）
- ✅ 支持自定义批次间延迟（5-300秒）
- ✅ 实时倒计时显示
- ✅ 支持中途停止
- ✅ 详细的日志输出
- ✅ 兼容原有行为（默认禁用分批）

### 3. 通用分批执行辅助函数

**位置**：在 `ensureConnection` 函数定义前添加

```javascript
// 通用分批执行辅助函数
const executeWithBatches = async (
  taskName,
  executeSingleToken,
  options = {}
) => {
  const {
    enableBatch = batchSettings.enableBatchExecution,
    batchSize = batchSettings.batchSize,
    batchDelay = batchSettings.batchDelay * 1000,
    maxRetries = 1,
    retryDelay = 3000,
  } = options;

  const totalTokens = selectedTokens.value.length;
  if (totalTokens === 0) return;

  const actualBatchSize = enableBatch ? batchSize : totalTokens;
  const totalBatches = Math.ceil(totalTokens / actualBatchSize);

  addLog({
    time: new Date().toLocaleTimeString(),
    message: `=== ${taskName}开始，共 ${totalTokens} 个账号${enableBatch ? `，分 ${totalBatches} 批执行，每批 ${actualBatchSize} 个账号` : ''} ===`,
    type: "info",
  });

  // 分批执行
  for (let batchIndex = 0; batchIndex < totalBatches; batchIndex++) {
    if (shouldStop.value) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== ${taskName}已停止 ===`,
        type: "warning",
      });
      break;
    }

    const startIdx = batchIndex * actualBatchSize;
    const endIdx = Math.min(startIdx + actualBatchSize, totalTokens);
    const batchTokens = selectedTokens.value.slice(startIdx, endIdx);

    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 开始执行第 ${batchIndex + 1}/${totalBatches} 批，共 ${batchTokens.length} 个账号 ===`,
      type: "info",
    });

    // 并行执行当前批次
    const batchPromises = batchTokens.map(async (tokenId) => {
      if (shouldStop.value) return;

      tokenStatus.value[tokenId] = "running";
      const token = tokens.value.find((t) => t.id === tokenId);

      let retryCount = 0;
      let success = false;

      while (retryCount <= maxRetries && !success) {
        if (shouldStop.value) break;

        try {
          await executeSingleToken(tokenId, token, retryCount);
          success = true;
          tokenStatus.value[tokenId] = "completed";
        } catch (error) {
          console.error(error);
          if (retryCount < maxRetries && !shouldStop.value) {
            addLog({
              time: new Date().toLocaleTimeString(),
              message: `${token.name} 执行出错: ${error.message}，等待${retryDelay/1000}秒后重试...`,
              type: "warning",
            });
            await new Promise((r) => setTimeout(r, retryDelay));
            retryCount++;
          } else {
            tokenStatus.value[tokenId] = "failed";
            addLog({
              time: new Date().toLocaleTimeString(),
              message: `${token.name} 执行失败: ${error.message}`,
              type: "error",
            });
          }
        }
      }
    });

    // 等待当前批次所有任务完成
    await Promise.all(batchPromises);

    // 如果不是最后一批且未停止，等待延迟时间
    if (batchIndex < totalBatches - 1 && !shouldStop.value && enableBatch) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== 第 ${batchIndex + 1} 批完成，等待 ${batchDelay/1000} 秒后执行下一批 ===`,
        type: "info",
      });

      // 倒计时显示
      const countdown = batchDelay / 1000;
      for (let i = countdown; i > 0 && !shouldStop.value; i--) {
        if (i % 5 === 0 || i <= 5) {
          addLog({
            time: new Date().toLocaleTimeString(),
            message: `距离下一批开始还有 ${i} 秒...`,
            type: "info",
          });
        }
        await new Promise((r) => setTimeout(r, 1000));
      }

      if (shouldStop.value) {
        addLog({
          time: new Date().toLocaleTimeString(),
          message: `=== 任务已停止，不再执行下一批 ===`,
          type: "warning",
        });
        break;
      }
    }
  }

  if (!shouldStop.value) {
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== ${taskName}全部完成 ===`,
      type: "success",
    });
  }
};
```

### 4. startBatch 函数修改

**位置**：替换原有的 `startBatch` 函数

```javascript
const startBatch = async () => {
  if (selectedTokens.value.length === 0) return;
  
  // 检查是否处于暂停时间
  if (isPauseTime.value.paused) {
    message.error(`当前处于${isPauseTime.value.reason}，任务暂停中，${pauseCountdown.value}后恢复`);
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 任务执行被阻止: 当前处于${isPauseTime.value.reason} ===`,
      type: "error",
    });
    return;
  }

  isRunning.value = true;
  shouldStop.value = false;

  // Reset status
  selectedTokens.value.forEach((id) => {
    tokenStatus.value[id] = "waiting";
  });

  const totalTokens = selectedTokens.value.length;
  const enableBatch = batchSettings.enableBatchExecution;
  const batchSize = enableBatch ? batchSettings.batchSize : totalTokens;
  const batchDelay = batchSettings.batchDelay * 1000;

  const totalBatches = Math.ceil(totalTokens / batchSize);

  addLog({
    time: new Date().toLocaleTimeString(),
    message: `=== 批量任务开始，共 ${totalTokens} 个账号${enableBatch ? `，分 ${totalBatches} 批执行，每批 ${batchSize} 个账号` : ''} ===`,
    type: "info",
  });

  // 分批执行
  for (let batchIndex = 0; batchIndex < totalBatches; batchIndex++) {
    if (shouldStop.value) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== 任务已停止 ===`,
        type: "warning",
      });
      break;
    }

    const startIdx = batchIndex * batchSize;
    const endIdx = Math.min(startIdx + batchSize, totalTokens);
    const batchTokens = selectedTokens.value.slice(startIdx, endIdx);

    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 开始执行第 ${batchIndex + 1}/${totalBatches} 批，共 ${batchTokens.length} 个账号 ===`,
      type: "info",
    });

    // 并行执行当前批次，但通过connectionQueue限制并发连接数
    const batchPromises = batchTokens.map(async (tokenId) => {
      if (shouldStop.value) return;

      tokenStatus.value[tokenId] = "running";

      let retryCount = 0;
      const MAX_RETRIES = 1;
      let success = false;

      while (retryCount <= MAX_RETRIES && !success) {
        if (shouldStop.value) break;

        const token = tokens.value.find((t) => t.id === tokenId);

        try {
          // 检查是否处于暂停时间
          if (isPauseTime.value.paused) {
            addLog({
              time: new Date().toLocaleTimeString(),
              message: `=== ${token.name} 任务被暂停: 当前处于${isPauseTime.value.reason}，已加入积攒队列 ===`,
              type: "info",
            });
            // 将任务加入积攒队列
            taskQueue.value.push({
              id: Date.now() + Math.random(),
              name: `批量任务-${token.name}`,
              runType: 'manual',
              selectedTokens: [tokenId],
              selectedTasks: selectedTasks.value,
            });
            success = true;
            tokenStatus.value[tokenId] = "completed";
            return;
          }

          if (retryCount === 0) {
            addLog({
              time: new Date().toLocaleTimeString(),
              message: `=== 开始执行: ${token.name} ===`,
              type: "info",
            });
          } else {
            addLog({
              time: new Date().toLocaleTimeString(),
              message: `=== 尝试重试: ${token.name} (第${retryCount}次) ===`,
              type: "info",
            });
          }

          await ensureConnection(tokenId);

          // Create runner with delay settings
          const runner = new DailyTaskRunner(tokenStore, {
            commandDelay: batchSettings.commandDelay,
            taskDelay: batchSettings.taskDelay,
          });

          // Run tasks
          await runner.run(tokenId, {
            onLog: (log) => addLog(log),
            onProgress: (p) => {
              // 每个token维护自己的进度
            },
          });

          success = true;
          tokenStatus.value[tokenId] = "completed";
          addLog({
            time: new Date().toLocaleTimeString(),
            message: `=== ${token.name} 执行完成 ===`,
            type: "success",
          });
        } catch (error) {
          console.error(error);
          if (retryCount < MAX_RETRIES && !shouldStop.value) {
            addLog({
              time: new Date().toLocaleTimeString(),
              message: `${token.name} 执行出错: ${error.message}，等待3秒后重试...`,
              type: "warning",
            });
            // Wait for potential token refresh in store
            await new Promise((r) => setTimeout(r, 3000));
            retryCount++;
          } else {
            tokenStatus.value[tokenId] = "failed";
            addLog({
              time: new Date().toLocaleTimeString(),
              message: `${token.name} 执行失败: ${error.message}`,
              type: "error",
            });
          }
        } finally {
          // 完成后关闭连接并释放槽位
          tokenStore.closeWebSocketConnection(tokenId);
          releaseConnectionSlot();
          addLog({
            time: new Date().toLocaleTimeString(),
            message: `${token.name} 连接已关闭  (队列: ${connectionQueue.active}/${batchSettings.maxActive})`,
            type: "info",
          });
        }
      }
    });

    // 等待当前批次所有任务完成
    await Promise.all(batchPromises);

    // 如果不是最后一批且未停止，等待延迟时间
    if (batchIndex < totalBatches - 1 && !shouldStop.value && enableBatch) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== 第 ${batchIndex + 1} 批完成，等待 ${batchSettings.batchDelay} 秒后执行下一批 ===`,
        type: "info",
      });

      // 倒计时显示
      const countdown = batchSettings.batchDelay;
      for (let i = countdown; i > 0 && !shouldStop.value; i--) {
        if (i % 5 === 0 || i <= 5) {
          addLog({
            time: new Date().toLocaleTimeString(),
            message: `距离下一批开始还有 ${i} 秒...`,
            type: "info",
          });
        }
        await new Promise((r) => setTimeout(r, 1000));
      }

      if (shouldStop.value) {
        addLog({
          time: new Date().toLocaleTimeString(),
          message: `=== 任务已停止，不再执行下一批 ===`,
          type: "warning",
        });
        break;
      }
    }
  }

  // 等待所有任务完成后再继续
  await new Promise((r) => setTimeout(r, 1000));

  isRunning.value = false;
  currentRunningTokenId.value = null;
  
  if (!shouldStop.value) {
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 批量任务全部完成 ===`,
      type: "success",
    });
    message.success("批量任务执行结束");
  }
};
```

## 使用示例

### 场景1：20个账号，每批5个，批次间延迟30秒

**配置**：
- 启用分批执行: ✅
- 每批账号数量: 5
- 批次间延迟: 30秒

**执行流程**：
```
=== 批量任务开始，共 20 个账号，分 4 批执行，每批 5 个账号 ===
=== 开始执行第 1/4 批，共 5 个账号 ===
  [账号1-5 并行执行...]
=== 第 1 批完成，等待 30 秒后执行下一批 ===
  [倒计时: 30, 25, 20, 15, 10, 5, 4, 3, 2, 1]
=== 开始执行第 2/4 批，共 5 个账号 ===
  [账号6-10 并行执行...]
=== 第 2 批完成，等待 30 秒后执行下一批 ===
  [倒计时...]
=== 开始执行第 3/4 批，共 5 个账号 ===
  [账号11-15 并行执行...]
=== 第 3 批完成，等待 30 秒后执行下一批 ===
  [倒计时...]
=== 开始执行第 4/4 批，共 5 个账号 ===
  [账号16-20 并行执行...]
=== 批量任务全部完成 ===
```

### 场景2：不使用分批（原有行为）

**配置**：
- 启用分批执行: ❌

**执行流程**：
```
=== 批量任务开始，共 20 个账号 ===
  [所有账号一次性执行，受maxActive限制]
=== 批量任务全部完成 ===
```

## 扩展其他批量任务

如果需要让其他批量任务也支持分批，可以使用 `executeWithBatches` 辅助函数：

```javascript
const batchXXX = async () => {
  isRunning.value = true;
  shouldStop.value = false;
  
  selectedTokens.value.forEach((id) => {
    tokenStatus.value[id] = "waiting";
  });

  await executeWithBatches("批量XXX", async (tokenId, token, retryCount) => {
    // 原有的单个账号执行逻辑
    await ensureConnection(tokenId);
    // ... 执行任务 ...
  });

  isRunning.value = false;
  message.success("批量XXX结束");
};
```

## 注意事项

1. **配置持久化**：配置会自动保存到 `localStorage`，刷新页面后保持
2. **兼容性**：默认禁用分批，不影响现有用户使用习惯
3. **并发控制**：批次内仍使用 `maxActive` 限制并发连接数
4. **停止功能**：支持在任意批次停止执行
5. **倒计时显示**：每5秒或最后5秒显示倒计时
