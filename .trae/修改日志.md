```javascript
          this.log("深海灯神今天已扫荡过（无免费次数），跳过", "info");
          return;
        }
        
        await this.executeGameCommand(
          tokenId,
          "genie_sweep",
          { genieId: 5, sweepCnt: 1 },
          "深海灯神",
        );
      } catch (e) {
        // 如果检查失败，尝试直接执行
        this.log(`检查深海灯神免费次数失败: ${e.message}，尝试直接执行`, "warning");
        await this.executeGameCommand(
          tokenId,
          "genie_sweep",
          { genieId: 5, sweepCnt: 1 },
          "深海灯神",
        );
      }
    },
  });
}
```

**执行流程：**
1. 周六/周日执行日常任务时，深海灯神任务加入队列
2. 执行前实时查询角色信息，检查是否还有免费次数
3. 有免费次数 → 正常执行扫荡
4. 无免费次数（已手动扫荡）→ 跳过，记录"深海灯神今天已扫荡过（无免费次数），跳过"
5. 查询失败 → 降级处理，尝试直接执行

## [2026-03-12] - 修复智能发车刷新太快导致识别不到颜色和大奖的问题

### 文件：`src/views/BatchDailyTasks.vue` 和 `src/utils/batch/tasksCar.js`

#### 修改类型（修复）

**问题描述：**
- 智能发车刷新速度太快（默认1000ms），服务器数据未及时更新
- 刷新后只依赖 `car_refresh` 接口返回的数据，没有重新查询确认
- 导致红车/大奖被误判为普通车，继续刷新浪费刷新券和金砖
- 用户反馈："日志哪里红车了还一直刷"、"刷新太快了，没识别到"

**修复方案（3个措施）：**

**1. 增加刷新延迟默认值**
- 将 `refreshDelay` 从 1000ms 增加到 1500ms
- 给服务器更多时间更新数据

**2. 刷新后重新查询车辆列表**
- 每次刷新后调用 `car_getrolecar` 重新获取完整车辆列表
- 从列表中找到当前车辆的最新数据
- 确保判断逻辑基于最新数据

**3. 增加额外确认延迟**
- 刷新后增加 500ms 额外延迟
- 确保服务器数据已完全更新

**关键代码：**

```javascript
// BatchDailyTasks.vue - 修改默认值
refreshDelay: 1500,       // 刷新延迟（发车刷新等）- 从1000ms增加到1500ms

// tasksCar.js - 刷新后重新查询车辆列表
const resp = await tokenStore.sendMessageWithPromise(
  tokenId,
  "car_refresh",
  { carId: String(car.id) },
  10000,
);

// 增加额外延迟，确保服务器数据已更新
await new Promise((r) => setTimeout(r, 500));

// 重新查询车辆列表获取最新数据
try {
  const carRes = await tokenStore.sendMessageWithPromise(
    tokenId,
    "car_getrolecar",
    {},
    10000,
  );
  const carList = normalizeCars(carRes?.body ?? carRes);
  const updatedCar = carList.find((c) => String(c.id) === String(car.id));
  if (updatedCar) {
    car.color = Number(updatedCar.color || 0);
    car.refreshCount = Number(updatedCar.refreshCount || 0);
    car.rewards = updatedCar.rewards || [];
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `${token.name} 刷新后车辆数据已更新: [${gradeLabel(car.color)}]`,
      type: "info",
    });
  }
} catch (e) {
  // 如果重新查询失败，使用刷新接口返回的数据（降级方案）
  const data = resp?.car || resp?.body?.car || resp;
  if (data && typeof data === "object") {
    if (data.color != null) car.color = Number(data.color);
    if (data.refreshCount != null)
      car.refreshCount = Number(data.refreshCount);
    if (data.rewards != null) car.rewards = data.rewards;
  }
  addLog({
    time: new Date().toLocaleTimeString(),
    message: `${token.name} 使用刷新接口返回数据: [${gradeLabel(car.color)}]`,
    type: "warning",
  });
}
```

**影响范围：**
- 修复了启用最大刷新次数限制时的刷新逻辑
- 修复了无次数限制时的刷新逻辑
- 两个刷新循环都增加了相同的保护措施
