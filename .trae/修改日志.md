```javascript
const BLACK_MARKET_CONFIG = {
  // 首次黑市周开启时间：2026年3月6日 12:00（周五）
  firstOpenTime: new Date('2026-03-06T12:00:00').getTime(),
  // 黑市周开启时长：6天12小时（周五12:00到下周五00:00）
  openDuration: 6 * 24 * 60 * 60 * 1000 + 12 * 60 * 60 * 1000,
  // 周期：3周（毫秒），每3周一次黑市周
  cycleDuration: 3 * 7 * 24 * 60 * 60 * 1000

// 计算当前是否处于黑市周开启期间
const isBlackMarketOpen = () => {

  const { firstOpenTime, cycleDuration, openDuration } = BLACK_MARKET_CONFIG;
```

## [2026-03-08] - 修复版本号递增逻辑

### 文件：`update-version.js`

#### 修改类型：修复

**问题描述：**
版本号更新逻辑有问题，导致版本号跳跃（如 3.23.2 → 3.23.4，跳过了 3.23.3）

**根本原因：**
原代码逻辑：
```javascript
if (patch > minor) {
  minor = minor + 1;
  patch = 0;
}
```
当 patch > minor 时，会同时增加 minor 并重置 patch，导致版本跳跃。

**修复方案：**
修改为正确的版本递增逻辑：
```javascript
// 递增修订版本号
patch = patch + 1;

// 当修订版本 >= 次版本时，次版本+1，修订版本归零
if (patch >= minor) {
  minor = minor + 1;
  patch = 0;
}

// 当次版本达到30时，主版本+1，次版本和修订版本归零
if (minor >= 30) {
  major = major + 1;
  minor = 0;
  patch = 0;
}
```

**效果：**
- 版本号每次只增加 1（如：3.23.16 → 3.23.17）
- 当修订版本达到次版本号时，次版本+1，修订版本归零
- 当次版本达到30时，主版本+1

- 导致任务冲突检测失效，多个任务可以并行执行

**修复方案：**
1. 修复 `executeInBatches` 函数的任务冲突检测逻辑
2. 添加 `isFromQueue` 参数，确保从队列中执行的任务不会再次加入队列
3. 修改 `checkAndExecuteQueuedTasks` 函数，使用原始任务函数执行子任务
4. 简化 `startBatch` 函数的任务冲突检测逻辑

**关键代码修改：**

```javascript

  // 检测任务冲突：有任务正在运行

    // 永远不会执行到这里
  }
}

// 修改后（修复）
const executeInBatches = async (taskFunction, taskName, taskFunctionName, isFromQueue = false) => {
  if (selectedTokens.value.length === 0) return;

  // 只有在任务不是从队列中执行时才检查冲突
  if (!isFromQueue) {

      // 检测到任务冲突，加入积攒队列依次执行
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== ${taskName} 检测到任务冲突（有任务运行中），加入积攒队列依次执行 ===`,
        type: "warning",
      });
      batchTaskStore.addToTaskQueue({
        id: Date.now() + Math.random(),
        name: taskName,
        runType: 'manual',
        selectedTokens: [...selectedTokens.value],
        selectedTasks: [taskFunctionName || taskName],
      });
      return;
    }

    // 确保任务状态正确
    batchTaskStore.startTask();
  }
};

// 修改 checkAndExecuteQueuedTasks 函数
for (const subTaskName of otherTasks) {
```
