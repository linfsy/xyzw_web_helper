### 3.1 队列长度限制（1.4）

**功能描述**：
限制队列长度，防止存储过大，最多保存50个任务。

**实现逻辑**：
1. 在保存队列时，限制队列长度
2. 超出限制时，保留最新的任务
3. 添加日志记录

**涉及文件**：
- `src/views/BatchDailyTasks.vue` - `saveTaskQueueToStorage` 函数

**代码实现**：

**文件：src/views/BatchDailyTasks.vue**
```javascript
const saveTaskQueueToStorage = () => {
  try {
    const queue = batchTaskStore.taskQueue || [];
    if (!Array.isArray(queue)) {
      console.error('任务队列不是数组:', queue);
      return;
    }
    
    // 限制队列长度，避免存储过大的数据
    const limitedQueue = queue.slice(0, 50); // 最多保存50个任务
    
    if (queue.length > 50) {
      addLog({
        time: new Date().toLocaleTimeString(),
        message: `=== 积攒队列任务过多(${queue.length}个)，只保留最新的50个 ===`,
        type: "warning",
      });
    }
    
    // 序列化前进行数据清理
    const sanitizedQueue = limitedQueue.map(task => {
      const sanitizedTask = { ...task };
      // 移除可能导致序列化问题的属性
      if (sanitizedTask.params) {
        sanitizedTask.params = {
          taskName: sanitizedTask.params.taskName,
          subTaskName: sanitizedTask.params.subTaskName,
        };
      }
      return sanitizedTask;
    });
    
    const dataToSave = JSON.stringify(sanitizedQueue);
    safeLocalStorage.setItem('batch_task_queue', dataToSave);
  } catch (error) {
    console.error('保存任务队列失败:', error);
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 保存积攒队列失败: ${error.message}，保留现有数据 ===`,
      type: "warning",
    });
  }
};
```

**测试方法**：
1. 添加超过50个任务到积攒队列
2. 保存队列到本地存储
3. 验证只保留了最新的50个任务
4. 验证日志中显示了警告信息

**注意事项**：
- 限制队列长度，避免存储空间不足
- 保留最新的任务，丢弃最旧的任务
- 添加日志记录，便于用户了解

**功能描述**：

```javascript
// 安全的 localStorage 操作封装
export const safeLocalStorage = {
  getItem(key) {
    try {
      return localStorage.getItem(key);
    } catch (error) {
      console.error(`localStorage getItem error for key "${key}":`, error);
      return null;
    }
  },
  
  setItem(key, value) {
    try {
      localStorage.setItem(key, value);
      return true;
    } catch (error) {
      console.error(`localStorage setItem error for key "${key}":`, error);
      
      // 如果是存储空间不足，尝试清理一些数据
      if (error.name === 'QuotaExceededError') {
        console.warn('Storage quota exceeded, attempting to clean up...');
        cleanupStorage();
        
        // 重试一次
        try {
          localStorage.setItem(key, value);
          return true;
        } catch (retryError) {
          console.error('Retry failed:', retryError);
          return false;
        }
      }
      
      return false;
    }
  },
  
  removeItem(key) {
    try {
      localStorage.removeItem(key);
      return true;
    } catch (error) {
      console.error(`localStorage removeItem error for key "${key}":`, error);
      return false;
    }
  },
  
  clear() {
    try {
      localStorage.clear();
      return true;
    } catch (error) {
      console.error('localStorage clear error:', error);
      return false;
    }
  }
};

// 清理存储空间
function cleanupStorage() {
  // 清理过期的日志
  const logKeys = Object.keys(localStorage).filter(key => key.startsWith('logs_'));
  logKeys.forEach(key => localStorage.removeItem(key));
  
  // 清理过期的任务标记
  const taskKeys = Object.keys(localStorage).filter(key => key.startsWith('task_executing_'));
  taskKeys.forEach(key => {
    const executionTime = parseInt(localStorage.getItem(key) || '0');
    if (Date.now() - executionTime > 30 * 60 * 1000) {
      localStorage.removeItem(key);
    }
  });
}
```

**使用方法**：
```javascript

localStorage.removeItem('key');

// 替换后

safeLocalStorage.setItem('key', value);
safeLocalStorage.getItem('key');
safeLocalStorage.removeItem('key');
```

const RUNNING_TASK_KEY = 'running_task_info';

// 保存运行中的任务信息
const saveRunningTask = (task) => {
  try {
    safeLocalStorage.setItem(RUNNING_TASK_KEY, JSON.stringify({
      ...task,
      timestamp: Date.now()
    }));
  } catch (e) {
    console.error('保存运行中任务失败:', e);
  }
};

  safeLocalStorage.removeItem(RUNNING_TASK_KEY);
};

// 恢复运行中的任务
const restoreRunningTask = () => {
  try {
    const taskInfo = safeLocalStorage.getItem(RUNNING_TASK_KEY);
    if (taskInfo) {
      const task = JSON.parse(taskInfo);
      // 检查任务是否过期（超过5分钟）
      if (Date.now() - (task.timestamp || 0) > 5 * 60 * 1000) {
        clearRunningTask();
        return null;
      }
      return task;
    }
  } catch (e) {
    console.error('恢复运行中任务失败:', e);
    clearRunningTask();
  }
  return null;
};

### 3.5 过期清理（3.3）

**功能描述**：
定期清理过期的执行标记和状态信息，避免存储空间浪费。

**实现逻辑**：
1. 清理过期的执行标记（超过30分钟）
2. 清理过期的任务执行时间标记（超过1小时）
3. 清理过期的运行中任务信息（超过5分钟）

**涉及文件**：
- `src/views/BatchDailyTasks.vue` - `onMounted` 函数
- `src/views/BatchDailyTasks.vue` - `cleanupLocalStorageTasks` 函数

**代码实现**：

**文件：src/views/BatchDailyTasks.vue**
```javascript
// 定期清理过期的 localStorage 数据
const cleanupLocalStorageTasks = () => {
  const now = new Date();
  
  Object.keys(localStorage).forEach(key => {
    try {
      if (key.startsWith('lastTaskExecution_')) {
        // 清理超过1小时的执行时间标记
        const executionTime = parseInt(localStorage.getItem(key) || '0');
        if (now.getTime() - executionTime > 60 * 60 * 1000) {
          localStorage.removeItem(key);
          console.log(`[Cleanup] 清理过期执行时间标记: ${key}`);
        }
      } else if (key.startsWith('task_executing_')) {
        // 清理超过30分钟的执行标记
        const executionTime = parseInt(localStorage.getItem(key) || '0');
        if (now.getTime() - executionTime > 30 * 60 * 1000) {
          localStorage.removeItem(key);
          console.log(`[Cleanup] 清理过期执行标记: ${key}`);
        }
      } else if (key === 'running_task_info') {
        // 清理超过5分钟的运行中任务信息
        const taskInfo = localStorage.getItem(key);
        if (taskInfo) {
          const task = JSON.parse(taskInfo);
          if (now.getTime() - (task.timestamp || 0) > 5 * 60 * 1000) {
            localStorage.removeItem(key);
            console.log(`[Cleanup] 清理过期运行中任务信息: ${key}`);
          }
        }
      }
    } catch (error) {
      console.warn(`[Cleanup] 清理 ${key} 时出错:`, error);
    }
  });
};

// 在组件挂载时启动清理定时器
onMounted(() => {
  // 立即执行一次清理
  cleanupLocalStorageTasks();
  
  // 每10分钟清理一次
  setInterval(cleanupLocalStorageTasks, 10 * 60 * 1000);
});
```

### 3.6 序列化安全（3.4）

**功能描述**：
处理循环引用和无法序列化的对象，确保数据可以安全序列化。

**实现逻辑**：
1. 序列化前清理可能导致循环引用的属性
2. 限制队列长度，避免存储过大
3. 捕获序列化异常，确保数据安全

**涉及文件**：
- `src/views/BatchDailyTasks.vue` - `saveTaskQueueToStorage` 函数

**代码实现**：

**文件：src/views/BatchDailyTasks.vue**
```javascript
const saveTaskQueueToStorage = () => {
  try {
    const queue = batchTaskStore.taskQueue || [];
    if (!Array.isArray(queue)) {
      console.error('任务队列不是数组:', queue);
      return;
    }
    
    // 限制队列长度，避免存储过大的数据
    const limitedQueue = queue.slice(0, 50);
    
    // 序列化前进行数据清理，移除可能导致循环引用的属性
    const sanitizedQueue = limitedQueue.map(task => {
      const sanitizedTask = { ...task };
      
      // 移除函数和复杂对象
      Object.keys(sanitizedTask).forEach(key => {
        const value = sanitizedTask[key];
        if (typeof value === 'function') {
          delete sanitizedTask[key];
        } else if (value instanceof Date) {
          sanitizedTask[key] = value.toISOString();
        } else if (value instanceof HTMLElement) {
          delete sanitizedTask[key];
        }
      });
      
      // 处理 params 属性，只保留简单数据类型
      if (sanitizedTask.params) {
        sanitizedTask.params = sanitizeParams(sanitizedTask.params);
      }
      
      return sanitizedTask;
    });
    
    const dataToSave = JSON.stringify(sanitizedQueue);
    safeLocalStorage.setItem('batch_task_queue', dataToSave);
  } catch (error) {
    console.error('保存任务队列失败:', error);
    addLog({
      time: new Date().toLocaleTimeString(),
      message: `=== 保存积攒队列失败: ${error.message}，保留现有数据 ===`,
      type: "warning",
    });
  }
};

// 清理参数，只保留可序列化的数据
const sanitizeParams = (params) => {
  if (!params || typeof params !== 'object') {
    return params;
  }
  
  const sanitized = {};
  Object.keys(params).forEach(key => {
    const value = params[key];
    if (typeof value === 'string' || 
        typeof value === 'number' || 
        typeof value === 'boolean' ||
        value === null ||
        (Array.isArray(value) && value.every(v => typeof v !== 'function')) ||
        (typeof value === 'object' && !(value instanceof HTMLElement))) {
      sanitized[key] = value;
    }
  });
  
  return sanitized;
};
```
