```typescript
const executeTask = async (task: ScheduledTask) => {
  // 防止重复执行
  if (executingTasks.value.has(task.id)) return;
  
  executingTasks.value.add(task.id);
  
  try {
    // 使用全局方法执行任务
    if (window['executeScheduledTask']) {
      await window['executeScheduledTask'](task);
    }
  } finally {
    executingTasks.value.delete(task.id);
  }
}
```

```typescript
const isPauseTime = computed(() => {
  const now = new Date();
  const dayOfWeek = now.getDay();
  const hours = now.getHours();
  const minutes = now.getMinutes();
  const minutesCount = hours * 60 + minutes;
  
  // 检查抢符时间
  const grabFuStart = timeToMinutes(batchSettings.grabFuStartTime);
  const grabFuEnd = timeToMinutes(batchSettings.grabFuEndTime);
  
  if (minutesCount >= grabFuStart && minutesCount <= grabFuEnd) {
    return { paused: true, reason: '抢符时间' };
  }
  
  // 检查周六盐场时间
  if (dayOfWeek === 6) {
    const saturdayStart = timeToMinutes(batchSettings.saturdaySaltStartTime);
    const saturdayEnd = timeToMinutes(batchSettings.saturdaySaltEndTime);
    
    if (minutesCount >= saturdayStart && minutesCount <= saturdayEnd) {
      return { paused: true, reason: '周六盐场' };
    }
  }
  
  // 检查周日盐场时间
  if (dayOfWeek === 0) {
    const sundayStart = timeToMinutes(batchSettings.sundaySaltStartTime);
    const sundayEnd = timeToMinutes(batchSettings.sundaySaltEndTime);
    
    if (minutesCount >= sundayStart && minutesCount <= sundayEnd) {
      return { paused: true, reason: '周日盐场' };
    }
  }
  
  return { paused: false, reason: '' };
});
```

```javascript
/**
 * 计算Cron表达式的下次执行时间
 * @param {Object} task - 任务对象
 * @returns {Date} 下次执行时间
 */
export function calculateNextExecutionTime(task) {
  const cronExpression = task.cronExpression;
  if (!cronExpression) return null;
  
  const parts = cronExpression.split(' ');
  if (parts.length !== 5) return null;
  
  const [minute, hour, dayOfMonth, month, dayOfWeek] = parts;
  
  // 解析各个字段
  const possibleMinutes = parseCronField(minute, 0, 59);
  const possibleHours = parseCronField(hour, 0, 23);
  const possibleDaysOfMonth = parseCronField(dayOfMonth, 1, 31);
  const possibleMonths = parseCronField(month, 1, 12);
  const possibleDaysOfWeek = parseCronField(dayOfWeek, 0, 6);
  
  // 计算下次执行时间
  let nextRun = new Date();
  nextRun.setSeconds(0, 0);
  
  // 优化的时间检查逻辑
  for (let i = 0; i < 366 * 24 * 60; i++) {
    const minutes = nextRun.getMinutes();
    const hours = nextRun.getHours();
    const date = nextRun.getDate();
    const month = nextRun.getMonth() + 1;
    const dayOfWeek = nextRun.getDay();
    
    // 1. 检查月份
    if (!possibleMonths.includes(month)) {
      nextRun.setMonth(nextRun.getMonth() + 1);
      nextRun.setDate(1);
      nextRun.setHours(0);
      nextRun.setMinutes(0);
      continue;
    }
    
    // 2. 检查日期和星期
    const matchesDayOfMonth = possibleDaysOfMonth.includes(date);
    const matchesDayOfWeek = possibleDaysOfWeek.includes(dayOfWeek);
    const matchesDay = dayOfMonth === '*' || dayOfWeek === '*' 
      ? matchesDayOfMonth || matchesDayOfWeek 
      : matchesDayOfMonth && matchesDayOfWeek;
    
    if (!matchesDay) {
      nextRun.setDate(nextRun.getDate() + 1);
      nextRun.setHours(0);
      nextRun.setMinutes(0);
      continue;
    }
    
    // 3. 检查小时
    if (!possibleHours.includes(hours)) {
      nextRun.setHours(nextRun.getHours() + 1);
      nextRun.setMinutes(0);
      continue;
    }
    
    // 4. 检查分钟
    if (possibleMinutes.includes(minutes)) {
      return nextRun;
    }
    
    nextRun.setMinutes(minutes + 1);
  }
  
  return null;
}
```
